def load_image(filename, width, height):
    """
    Loads and scales an image from assets folder
    Arguments: 
      filename (string) - name of image file
      width (int) - desired width in pixels
      height (int) - desired height in pixels
    Returns: Pygame Surface object with scaled image, or None if loading fails

        # pygame.image.load(path) returns: Surface object with loaded image
        image = pygame.image.load(f"assets/{filename}")
        # pygame.transform.scale(surface, (width, height)) returns: New scaled Surface

        return image
    except:
        # Error handling if image file not found
        print(f"Could not load image: {filename}")
        return None

# PLAYER UPDATE FUNCTION - MOVEMENT AND PHYSICS
# =============================================
def update_player():
    """
    Updates player position and physics every frame
    Handles jumping gravity and lane movement
    """
    # Declare global variables we want to modify
    global player_y, player_vel_y, player_jumping, player_x
    
    # JUMPING PHYSICS - only apply when player is jumping
    if player_jumping:
        # Add gravity to vertical velocity (makes player fall faster)
        player_vel_y += GRAVITY
        # Update Y position based on velocity
        player_y += player_vel_y
    
    # GROUND COLLISION DETECTION - stop falling when hit ground
    if player_jumping and player_y >= SCREEN_HEIGHT - RUNNER_HEIGHT - 20:
        player_y = SCREEN_HEIGHT - RUNNER_HEIGHT - 20  # Snap to ground position
        player_vel_y = 0      # Reset velocity
        player_jumping = False  # No longer jumping
    
    # SMOOTH LANE MOVEMENT - move gradually toward target lane
    target_x = LANES[player_lane]  # Get target X position for current lane
    # Move 20% of remaining distance each frame for smooth animation
    player_x += (target_x - player_x) * 0.2

# PLAYER DRAWING FUNCTION - VISUAL REPRESENTATION
# =============================================
def draw_player():
    """
    Draws the player character in correct state and position
    Uses images if available, otherwise draws colored rectangles
    """
    if player_crouching and player_crouch_img:
        # Draw crouching player - positioned lower because height is halved
        # screen.blit(source, dest) returns: None
        # Arguments: source (Surface) - image to draw, dest (x,y) - position to draw at
        screen.blit(player_crouch_img, (player_x - RUNNER_WIDTH//2, player_y + RUNNER_HEIGHT//2))
    elif player_jumping and player_jump_img:
        # Draw jumping player - normal position but jump pose
        screen.blit(player_jump_img, (player_x - RUNNER_WIDTH//2, player_y))
    elif player_normal_img:
        # Draw normal running player
        screen.blit(player_normal_img, (player_x - RUNNER_WIDTH//2, player_y))
    else:
        # FALLBACK: Draw colored rectangle if images not available
        color = BLUE if not player_jumping else (100, 200, 255)  # Light blue when jumping
        height = RUNNER_HEIGHT if not player_crouching else RUNNER_HEIGHT//2
        y_pos = player_y if not player_crouching else player_y + RUNNER_HEIGHT//2
        
        # pygame.draw.rect(surface, color, rect) returns: None  
        # Arguments: surface - where to draw, color - fill color, rect - (x,y,width,height)
        pygame.draw.rect(screen, color, (player_x - RUNNER_WIDTH//2, y_pos, RUNNER_WIDTH, height))
        # Draw white outline around player
        pygame.draw.rect(screen, WHITE, (player_x - RUNNER_WIDTH//2, y_pos, RUNNER_WIDTH, height), 2)

# OBSTACLE CREATION FUNCTION
# =============================================
def create_obstacle():
    """
    Creates a new obstacle with random type and lane
    Adds it to the obstacles list
    """
    lane = random.randint(0, 2)  # Random lane: 0=left, 1=center, 2=right
    # Randomly choose obstacle type
    obstacle_type = random.choice(['box', 'barrier', 'hole', 'train', 'low_barrier'])
    
    # Set properties based on obstacle type
    if obstacle_type == 'box':
        width, height = 60, 60
        image = box_img
    elif obstacle_type == 'barrier':
        width, height = 80, 80
        image = barrier_img
    elif obstacle_type == 'hole':
        width, height = 100, 40
        image = hole_img
    elif obstacle_type == 'train':
        width, height = 140, 120
        image = train_img
    else:  # low_barrier
        width, height = 70, 35
        image = low_barrier_img
    
    # Create obstacle dictionary with all properties
    obstacles.append({
        'x': LANES[lane],      # X position based on lane
        'y': -height,          # Start above screen (negative Y)
        'width': width,        # Obstacle width
        'height': height,      # Obstacle height  
        'type': obstacle_type, # Type for collision logic
        'image': image,        # Image for drawing
        'lane': lane,          # Which lane it's in
        'speed': obstacle_speed # Current movement speed
    })

# OBSTACLE DRAWING FUNCTION
# =============================================
def draw_obstacles():
    """
    Draws all active obstacles on screen
    Uses images if available, otherwise draws colored rectangles
    """
    for obstacle in obstacles:
        if obstacle['image']:
            # Draw obstacle image centered at its position
            screen.blit(obstacle['image'], 
                       (obstacle['x'] - obstacle['width']//2, obstacle['y']))
        else:
            # FALLBACK: Draw colored rectangle based on obstacle type
            if obstacle['type'] == 'box':
                pygame.draw.rect(screen, RED, (obstacle['x'] - obstacle['width']//2, obstacle['y'], obstacle['width'], obstacle['height']))
            elif obstacle['type'] == 'barrier':
                pygame.draw.rect(screen, (200, 100, 0), (obstacle['x'] - obstacle['width']//2, obstacle['y'], obstacle['width'], obstacle['height']))
            elif obstacle['type'] == 'hole':
                pygame.draw.rect(screen, BLACK, (obstacle['x'] - obstacle['width']//2, obstacle['y'], obstacle['width'], obstacle['height']))
            elif obstacle['type'] == 'train':
                pygame.draw.rect(screen, (60, 60, 60), (obstacle['x'] - obstacle['width']//2, obstacle['y'], obstacle['width'], obstacle['height']))
            else:  # low_barrier
                pygame.draw.rect(screen, (255, 165, 0), (obstacle['x'] - obstacle['width']//2, obstacle['y'], obstacle['width'], obstacle['height']))

# OBSTACLE UPDATE FUNCTION - MOVEMENT
# =============================================
def update_obstacles():
    """
    Moves all obstacles downward and removes off-screen ones
    """
    global obstacles
    # Loop through all obstacles (use [:] to iterate over copy for safe removal)
    for obstacle in obstacles[:]:
        # Move obstacle downward based on its speed
        obstacle['y'] += obstacle['speed']
        # Check if obstacle has moved off bottom of screen
        if obstacle['y'] > SCREEN_HEIGHT + 50:
            obstacles.remove(obstacle)  # Remove from active obstacles list

# COLLISION DETECTION FUNCTION
# =============================================
def check_collisions():
    """
    Checks if player collides with any obstacle
    Returns: True if collision detected, False otherwise
    Special rules for holes (must jump) and low barriers (must crouch)
    """
    # Adjust player height and position based on crouching state
    player_height_temp = RUNNER_HEIGHT if not player_crouching else RUNNER_HEIGHT//2
    player_y_temp = player_y if not player_crouching else player_y + RUNNER_HEIGHT//2
    
    # Create rectangle for player collision detection
    # pygame.Rect(x, y, width, height) returns: Rect object
    player_rect = pygame.Rect(player_x - RUNNER_WIDTH//2, player_y_temp, RUNNER_WIDTH, player_height_temp)
    
    # Check collision with each obstacle
    for obstacle in obstacles:
        # Create rectangle for obstacle
        obstacle_rect = pygame.Rect(obstacle['x'] - obstacle['width']//2, 
                                  obstacle['y'], 
                                  obstacle['width'], 
                                  obstacle['height'])
        
        # rect.colliderect(other_rect) returns: True if rectangles overlap
        if player_rect.colliderect(obstacle_rect):
            # SPECIAL COLLISION RULES:
            if obstacle['type'] == 'hole':
                # Only die if not jumping over hole
                if not player_jumping:  
                    return True
            elif obstacle['type'] == 'low_barrier':
                # Only die if not crouching under low barrier
                if not player_crouching:  
                    return True
            else:
                # Die for all other obstacle types
                return True  
    return False  # No collisions detected

# TRACK DRAWING FUNCTION - BACKGROUND
# =============================================
def draw_tracks():
    """
    Draws the scrolling track background for all lanes
    Creates illusion of forward movement
    """
    global track_offset
    # Update scroll offset and wrap around using modulo
    track_offset = (track_offset + track_speed) % 40
    
    # Draw tracks for each lane
    for lane in LANES:
        if use_track_image:
            # Use track image with vertical tiling and scrolling
            for y in range(-40, SCREEN_HEIGHT + 40, 40):
                track_y = y + track_offset  # Apply scrolling offset
                # image.get_rect(topleft=(x,y)) returns: Rect positioned at (x,y)
                track_rect = track_img.get_rect(topleft=(lane - 80, track_y))
                screen.blit(track_img, track_rect)
        else:
            # FALLBACK: Draw procedural tracks (colored rectangles)
            pygame.draw.rect(screen, TRACK_COLOR, (lane - 80, 0, 160, SCREEN_HEIGHT))
            
            # Draw moving track lines
            for y in range(-40, SCREEN_HEIGHT + 40, 40):
                line_y = y + track_offset
                pygame.draw.rect(screen, YELLOW, (lane - 60, line_y, 120, 10))
        
        # Draw lane borders (always visible)
        pygame.draw.rect(screen, (150, 150, 150), (lane - 80, 0, 10, SCREEN_HEIGHT))
        pygame.draw.rect(screen, (150, 150, 150), (lane + 70, 0, 10, SCREEN_HEIGHT))

def reset_game():
    """
    Resets all game variables to start a new game
    """
    global player_x, player_y, player_vel_y, player_jumping, player_crouching, player_lane
    global obstacles, score, obstacle_speed, obstacle_timer, track_offset, track_speed
    global last_speed_increase_score, elapsed_time
    
    # Reset player to starting position and state
    player_x = LANES[1]
    player_y = SCREEN_HEIGHT - RUNNER_HEIGHT - 20
    player_vel_y = 0
    player_jumping = False
    player_crouching = False
    player_lane = 1
    
    # Clear all obstacles and reset game progress
    obstacles = []
    score = 0
    obstacle_speed = BASE_OBSTACLE_SPEED
    track_speed = BASE_TRACK_SPEED
    obstacle_timer = 0
    track_offset = 0
    last_speed_increase_score = 0
    elapsed_time = 0  # Reset scoring timer

# =============================================
# GESTURE HANDLING FUNCTION
# =============================================
def handle_gesture(gesture):
    """
    Converts hand gestures to player actions
    Arguments: gesture (string) - detected hand gesture
    """
    global current_gesture, player_jumping, player_crouching, player_lane, player_vel_y
    
    current_gesture = gesture  # Store current gesture for display
    
    # Only process gestures during gameplay
    if game_state != "playing":
        return
    
    # Map gestures to player actions
    if gesture == "jump" and not player_jumping and not player_crouching:
        # Start jump if not already jumping or crouching
        player_vel_y = JUMP_STRENGTH
        player_jumping = True
        player_crouching = False
        print("JUMP!")  # Debug output
    elif gesture == "crouch" and not player_jumping:
        # Crouch if not jumping
        player_crouching = True
    elif gesture == "left" and player_lane > 0:
        # Move left if not already in leftmost lane
        player_lane -= 1
    elif gesture == "right" and player_lane < 2:
        # Move right if not already in rightmost lane  
        player_lane += 1
    elif gesture == "none":
        # Stop crouching when hand returns to center
        player_crouching = False

# =============================================
# SCORING SYSTEM FUNCTION
# =============================================
def update_score():
    """
    Updates score continuously based on game speed
    Called every frame to add points over time
    """
    global score, elapsed_time
    elapsed_time += 1/FPS  # Add time since last frame (in seconds)
    
    # Add points every 0.1 seconds
    if elapsed_time >= 0.1:
        # Points = current speed × 2 (faster speed = more points)
        points_to_add = int(obstacle_speed * 2)
        score += points_to_add
        elapsed_time = 0  # Reset timer

# =============================================
# DIFFICULTY INCREASE FUNCTION
# =============================================
def increase_speed():
    """
    Increases game speed progressively as score increases
    Makes game harder over time
    """
    global obstacle_speed, track_speed, last_speed_increase_score
    # Increase speed every 50 points, but only once per 50-point milestone
    if score > 0 and score % 50 == 0 and score != last_speed_increase_score:
        obstacle_speed += 1.0  # Make obstacles faster
        track_speed += 1.0     # Make track scroll faster
        last_speed_increase_score = score  # Remember we increased at this score
        print(f"🚀 Speed increased to: {obstacle_speed}")  # Debug output

# =============================================
# MAIN GAME LOOP - HEART OF THE GAME
# =============================================
running = True  # Control variable for main loop
print("Game started! Press SPACE to begin...")

# MAIN GAME LOOP - runs 60 times per second
while running:
    # Control game speed - wait to maintain 60 FPS
    # clock.tick(FPS) returns: milliseconds since last call
    clock.tick(FPS)
    
    # =========================================
    # INPUT HANDLING SECTION
    # =========================================
    
    # Get hand gesture from camera
    if hand_controller:
        # hand_controller.get_gesture() returns: string with gesture name
        gesture = hand_controller.get_gesture()
        if gesture == "quit":  # Quit if 'q' pressed in camera window
            running = False
    else:
        gesture = "none"  # Default if hand controller not available
    
    # Handle keyboard and window events
    # pygame.event.get() returns: list of Event objects
    for event in pygame.event.get():
        if event.type == pygame.QUIT:  # Window close button clicked
            running = False
        
        # Handle keyboard input
        if event.type == pygame.KEYDOWN:  # Key pressed event
            if event.key == pygame.K_SPACE and game_state == "menu":
                # Start game from menu
                game_state = "playing"
                reset_game()
                print("Game started! Use hand gestures to control.")
            elif event.key == pygame.K_r and game_state == "game_over":
                # Restart game after game over
                game_state = "playing"
                reset_game()
            elif event.key == pygame.K_q:  # Quit game
                running = False
    
    # =========================================
    # GAME LOGIC SECTION
    # =========================================
    
    # Process hand gestures
    handle_gesture(gesture)
    
    # Only update game logic when playing (not in menu/game over)
    if game_state == "playing":
        # Update player position and physics
        update_player()
        
        # Update score continuously
        update_score()
        
        # Spawn new obstacles at intervals
        obstacle_timer += 1
        if obstacle_timer >= OBSTACLE_SPAWN_INTERVAL:
            create_obstacle()
            obstacle_timer = 0  # Reset spawn timer
        
        # Move all existing obstacles
        update_obstacles()
        
        # Increase game speed as score increases
        increase_speed()
        
        # Check for collisions with obstacles
        if check_collisions():
            game_state = "game_over"  # End game if collision detected
            print("Game Over! Collision detected.")
    
    # =========================================
    # RENDERING SECTION - DRAW EVERYTHING
    # =========================================
    
    # Clear screen with background color
    # surface.fill(color) returns: None
    screen.fill(GRAY)
    
    # Draw game elements in correct order (background to foreground)
    draw_tracks()      # Background tracks
    draw_obstacles()   # Obstacles (on top of tracks)
    draw_player()      # Player (on top of obstacles)
    draw_score()       # UI text (on top of everything)
    
    # Draw appropriate screen based on game state
    if game_state == "menu":
        draw_menu()
    elif game_state == "game_over":
        draw_game_over()
    
    # Update display to show everything we've drawn
    # pygame.display.flip() returns: None
    # Makes all drawing operations visible on screen
    pygame.display.flip()

# =============================================
# CLEANUP SECTION - PROPERLY CLOSE EVERYTHING
# =============================================

# Close hand controller and camera
if hand_controller:
    hand_controller.cleanup()

# Close Pygame and free resources
# pygame.quit() returns: None
pygame.quit()
